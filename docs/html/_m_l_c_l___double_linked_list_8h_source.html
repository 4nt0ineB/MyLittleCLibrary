35 /**
36  * @brief Descriptor of the linked list
37  * The list - the cells - refers to a double linked list descriptor,
38  * gathering implemented functions of the DoubleLinkedList type, including
39  * a TypeDescriptor allowing manipulation of the data, and the list length.
40  */
41 typedef struct s_double_linked_list_descriptor {
42  TypeDescriptor * type_descriptor;
43  int length; /*<! Linked list length */
44  int (*add_) (DoubleLinkedList *, const void *, int (*cmp) (const void *, const void *)); /**< ordered add depending on given function. Must return 1 or 0 */
45  int (*ordered_add) (DoubleLinkedList *, const void *); /**< Ascending add */
46  int (*reverse_ordered_add) (DoubleLinkedList *, const void *); /**< Descending add*/
47  int (*prepend) (DoubleLinkedList *, const void *); /**< Add data to the head */
48  int (*append) (DoubleLinkedList *, const void *); /**< Add data to the tail*/
49  int (*insert) (DoubleLinkedList *, const void *); /**< Insert given data right next to the given cell */
50  DoubleLinkedCell * (*search) (DoubleLinkedList, const void *); /**< Search and return a pointer to the first found occurrence of the given data */
51  void * (*extract) (DoubleLinkedCell **);
52  int (*remove) (DoubleLinkedList *, const void *); /**< Search and remove first occurrence of the given data */
53  /* @Todo filter_remove */
54  /* @Todo cell_remove */
55  void * (*shift) (DoubleLinkedList *); /**< Remove head's cell and return the pointer to its data */
56  void * (*pop) (DoubleLinkedList *); /**< Remove tail's cell and return the pointer to its data */
57  void (*cell_free) (DoubleLinkedCell **); /**< Free the given cell */
58  void (*free) (DoubleLinkedList *); /**< Free the list */
59  void (*cell_print) (DoubleLinkedList); /**< Print the given cell on stdout */
60  void (*cell_fprint) (FILE *, DoubleLinkedCell *); /**< Print the given cell on given stream */
61  void (*print) (DoubleLinkedList); /**< Print the list on stdout */
62  void (*fprint) (FILE *, DoubleLinkedList); /**< Print the list on the given stream*/
63  void (*to_dot) (DoubleLinkedList, const char * path); /**< Make a formatted dot file of the struct */
64 } DoubleLinkedListDescriptor;
