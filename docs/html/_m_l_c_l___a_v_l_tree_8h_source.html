34 /**
35  * @brief Descriptor of the AVL tree
36  * The tree - the nodes - refers to a binary tree descriptor,
37  * gathering implemented functions of the BinarySearchTree type, including
38  * a TypeDescriptor allowing manipulation of the data, and the the number of nodes.
39  */
40 typedef struct s_avl_tree_descriptor {
41  TypeDescriptor * type_descriptor;
42  int n; /**< Number of nodes in the tree */
43  int (*height) (AVLTree); /**< Return the height of the tree */
44  int (*nb_nodes) (AVLTree); /**< Return the number of nodes. O(1) by the struct */
45  int (*nb_leaves) (AVLTree); /**< Return the number of leaves */
46  int (*nb_internal_nodes) (AVLTree); /**< Return the number of internal nodes */
47  int (*nb_two_children) (AVLTree); /**< Number of nodes having two children */
48  int (*add) (AVLTree *, const void *); /**< add data to the tree */
49  AVLTree (*min) (AVLTree); /**< Return a pointer to the min node */
50  AVLTree (*max) (AVLTree); /**< Return a pointer to the the max node */
51  int (*extract_min) (AVLTree *, void *); /**< Remove the min node in the tree and return the pointer of its data */
52  int (*extract_max) (AVLTree *, void *); /**< Remove the max node in the tree and return the pointer of its data */
53  int (*remove) (AVLTree *, const void *); /**< Remove the first occurrence of the given data trough preorder traversal */
54  AVLTree (*search) (AVLTree *, const void *); /**< Return a pointer to the first occurrence of the node having data equal to the given one, trough preorder traversal */
55  int (*is_perfect_bt) (AVLTree); /**< Return 1 if the tree is a perfect binary tree, else 0 */
56  int (*is_bst) (AVLTree); /**< Return 1 if the tree is a binary search tree, else 0 */
57  void (*node_print) (AVLTreeNode *); /**< Print the given node on stdout */
58  void (*node_fprint) (FILE *, AVLTreeNode *); /**< Print the given node on the given stream */
59  void (*fprint_preorder) (FILE *, AVLTree);
60  void (*fprint_inorder) (FILE *, AVLTree);
61  void (*fprint_postorder) (FILE *, AVLTree);
62  void (*to_dot) (AVLTree, const char *); /**< Make a formatted dot file of the struct */
63  void (*free) (AVLTree *); /**< Free to tree */
64 } AVLTreeDescriptor;
