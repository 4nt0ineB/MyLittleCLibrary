36 /**
37  * @brief Descriptor of the binary tree
38  * The tree - the nodes - refers to a binary tree descriptor,
39  * gathering implemented functions of the BinarySearchTree type, including
40  * a TypeDescriptor allowing manipulation of the data, and the the number of nodes.
41  */
42 typedef struct s_binary_search_tree_descriptor {
43  int n; /**< Number of nodes in the tree */
44  TypeDescriptor * type_descriptor;
45  int (*height) (BinarySearchTree); /**< Return the height of the tree */
46  int (*nb_nodes) (BinarySearchTree); /**< Return the number of nodes. O(1) by the struct */
47  int (*nb_leaves) (BinarySearchTree); /**< Return the number of leaves */
48  int (*nb_internal_nodes) (BinarySearchTree); /**< Return the number of internal nodes */
49  int (*nb_two_children) (BinarySearchTree); /**< Number of nodes having two children */
50  int (*add) (BinarySearchTree *, const void *); /**< add data to the tree */
51  BinarySearchTree (*min) (BinarySearchTree); /**< Return a pointer to the min node */
52  BinarySearchTree (*max) (BinarySearchTree); /**< Return a pointer to the the max node */
53  void * (*extract_min) (BinarySearchTree *); /**< Remove the min node in the tree and return the pointer of its data */
54  void * (*extract_max) (BinarySearchTree *); /**< Remove the max node in the tree and return the pointer of its data */
55  int (*remove) (BinarySearchTree *, const void *); /**< Remove the first occurrence of the given data trough preorder traversal */
56  BinarySearchTree (*search) (BinarySearchTree *, const void *); /**< Return a pointer to the first occurrence of the node having data equal to the given one, trough preorder traversal */
57  int (*is_perfect_bt) (BinarySearchTree); /**< Return 1 if the tree is a perfect binary tree, else 0 */
58  int (*is_bst) (BinarySearchTree); /**< Return 1 if the tree is a binary search tree, else 0 */
59  void (*node_print) (BinarySearchTreeNode *); /**< Print the given node on stdout */
60  void (*node_fprint) (FILE *, BinarySearchTreeNode *); /**< Print the given node on the given stream */
61  void (*fprint_preorder) (FILE *, BinarySearchTree);
62  void (*fprint_inorder) (FILE *, BinarySearchTree);
63  void (*fprint_postorder) (FILE *, BinarySearchTree);
64  /* breadth-first traversal */
65  void (*to_dot) (BinarySearchTree, const char *); /**< Make a formatted dot file of the struct */
66  void (*free) (BinarySearchTree *); /**< Free to tree */
67 } BinarySearchTreeDescriptor;
